175    async def fetch(
176            self,
177            url: str,
178            method: str = "GET",
179            params: Optional[Dict[str, Any]] = None,
180            data: Any = None,
181            json: Any = None,
182            headers: Optional[Dict[str, str]] = None,
183            content_type: Optional[str] = None,
184            timeout: Optional[int] = None,
185            retry_count: int = 0
186    ) -> Any:
187        """Make an authenticated HTTP request.
188        
189        This method handles authentication, retries, error handling, and response parsing.
190        
191        Args:
192            url: The URL to request
193            method: HTTP method to use. Defaults to "GET".
194            params: Query parameters
195            data: Request body data
196            json: JSON data to send (will set content_type to application/json)
197            headers: Additional HTTP headers
198            content_type: Content-Type header
199            timeout: Request timeout in seconds
200            retry_count: Current retry attempt (used internally)
201            
202        Returns:
203            Response data, parsed as JSON if possible
204            
205        Raises:
206            HTTPError: For HTTP error responses
207            RateLimitError: When rate limited by the API
208            Exception: For other request errors
209        """
210        if not self._session:
211            self._session = aiohttp.ClientSession()
212
213        # Prepare request
214        if json is not None:
215            data = json
216            content_type = "application/json"
217
218        final_headers = {}
219        
220        if self.auth and "Authorization" not in (headers or {}):
221            auth_type = AuthType(self.auth.get("auth_type", "PlatformOauth2"))
222            credentials = self.auth.get("credentials", {})
223            
224            if auth_type == AuthType.PlatformOauth2 and "access_token" in credentials:
225                final_headers["Authorization"] = f"Bearer {credentials['access_token']}"
226
227        if content_type:
228            final_headers["Content-Type"] = content_type
229        if headers:
230            final_headers.update(headers)
231
232        if params:
233            # Handle nested dictionary parameters
234            flat_params = {}
235            for key, value in params.items():
236                if isinstance(value, (dict, list)):
237                    flat_params[key] = jsonX.dumps(value)
238                elif value is not None:
239                    flat_params[key] = str(value)
240            query_string = urlencode(flat_params)
241            url = f"{url}{'&' if '?' in url else '?'}{query_string}"
242
243        # Prepare body
244        if data is not None:
245            if content_type == "application/json":
246                data = jsonX.dumps(data)
247            elif content_type == "application/x-www-form-urlencoded":
248                data = urlencode(data) if isinstance(data, dict) else data
249
250        # Store the original timeout numeric value
251        original_timeout = timeout or self.config["timeout"]
252
253        # Convert the numeric timeout to a ClientTimeout instance for this request
254        client_timeout = aiohttp.ClientTimeout(total=original_timeout)
255
256        try:
257            async with self._session.request(
258                method=method,
259                url=url,
260                data=data,
261                headers=final_headers,
262                timeout=client_timeout,
263                ssl=True
264            ) as response:
265                content_type = response.headers.get("Content-Type", "")
266
267                if response.status == 429:  # Rate limit
268                    retry_after = int(response.headers.get("Retry-After", 60))
269                    raise RateLimitError(
270                        retry_after,
271                        response.status,
272                        "Rate limit exceeded",
273                        await response.text()
274                    )
275
276                try:
277                    if "application/json" in content_type:
278                        result = await response.json()
279                    else:
280                        result = await response.text()
281                        if not result and response.status in {200, 201, 204}:
282                            return None
283                except Exception as e:
284                    self.logger.error(f"Error parsing response: {e}")
285                    result = await response.text()
286
287                if not response.ok:
288                    print(f"HTTP error encountered. Status: {response.status}. Result: {result}")
289                    raise HTTPError(response.status, str(result), result)
290
291                return result
292
293        except RateLimitError:
294            raise
295        except (aiohttp.ClientError, asyncio.TimeoutError) as e:
296            print(f"Error encountered: {e}. Retry count: {retry_count}. Backing off.")
297            if retry_count < self.config["max_retries"]:
298                await asyncio.sleep(2 ** retry_count)  # Exponential backoff
299                print("Retrying request...")
300                # Use original_timeout (numeric) for recursive calls
301                return await self.fetch(
302                    url, method, params, data, json,
303                    headers, content_type, original_timeout, retry_count + 1
304                )
305            else:
306                print("Max retries reached. Raising error.")
307                raise
308        except Exception as e:
309            self.logger.error(f"Unexpected error during {method} {url}: {e}")
310            print(f"Unexpected error encountered: {e}")
311            raise

530    async def execute_action(self,
531                           name: str,
532                           inputs: Dict[str, Any],
533                           context: ExecutionContext) -> Any:
534        """Execute a registered action.
535        
536        Args:
537            name: Name of the action to execute
538            inputs: Action inputs
539            context: Execution context
540            
541        Returns:
542            Action result
543            
544        Raises:
545            ValidationError: If inputs or outputs don't match schema
546        """
547        if name not in self._action_handlers:
548            raise ValidationError(f"Action '{name}' not registered")
549
550        # Validate inputs against action schema
551        action_config = self.config.actions[name]
552        validator = Draft7Validator(action_config.input_schema)
553        errors = sorted(validator.iter_errors(inputs), key=lambda e: e.path)
554        if errors:
555            message = ""
556            for error in errors:
557                message += f"{list(error.schema_path)}, {error.message},\n "
558            raise ValidationError(message, action_config.input_schema, inputs)
559         
560        if "fields" in self.config.auth:
561            auth_config = self.config.auth["fields"]
562            validator = Draft7Validator(auth_config)
563            errors = sorted(validator.iter_errors(context.auth), key=lambda e: e.path)
564            if errors:
565                message = ""
566                for error in errors:
567                    message += f"{list(error.schema_path)}, {error.message},\n "
568                raise ValidationError(message, auth_config, context.auth)
569
570        # Create handler instance and execute
571        handler = self._action_handlers[name]()
572        result = await handler.execute(inputs, context)
573
574        # Validate output if schema is defined
575        validator = Draft7Validator(action_config.output_schema)
576        errors = sorted(validator.iter_errors(result), key=lambda e: e.path)
577        if errors:
578            message = ""
579            for error in errors:
580                message += f"{list(error.schema_path)}, {error.message},\n "
581            raise ValidationError(message, action_config.output_schema, result)
582     
583        return result

585    async def execute_polling_trigger(self,
586                                    name: str,
587                                    inputs: Dict[str, Any],
588                                    last_poll_ts: Optional[str],
589                                    context: ExecutionContext) -> List[Dict[str, Any]]:
590        """Execute a registered polling trigger
591        
592        Args:
593            name: Name of the polling trigger to execute
594            inputs: Trigger inputs
595            last_poll_ts: Last poll timestamp
596            context: Execution context
597            
598        Returns:
599            List of records
600            
601        Raises:
602            ValidationError: If inputs or outputs don't match schema
603        """
604        if name not in self._polling_handlers:
605            raise ValidationError(f"Polling trigger '{name}' not registered")
606
607        # Validate trigger configuration
608        trigger_config = self.config.polling_triggers[name]
609        try:
610            validate(inputs, trigger_config.input_schema)
611        except Exception as e:
612            raise ValidationError(e.message, e.schema, e.instance)
613
614        try:
615            auth_config = self.config.auth["fields"]
616            validate(context.auth, auth_config)
617        except Exception as e:
618            raise ValidationError(e.message, e.schema, e.instance)
619        
620        # Create handler instance and execute
621        handler = self._polling_handlers[name]()
622        records = await handler.poll(inputs, last_poll_ts, context)
623        # Validate each record
624        for record in records:
625            if "id" not in record:
626                raise ValidationError(
627                    f"Polling trigger '{name}' returned record without required 'id' field")
628            if "data" not in record:
629                raise ValidationError(
630                    f"Polling trigger '{name}' returned record without required 'data' field")
631
632            # Validate record data against output schema
633            try:
634                validate(record["data"], trigger_config.output_schema)
635            except Exception as e:
636                raise ValidationError(e.message, e.schema, e.instance)
637            
638        return records
